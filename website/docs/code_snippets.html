broadcast_excluding_player

Sends a chat message to all connected players except the specified one. Useful for admin announcements or private acknowledgements.

Parameters:

  • message string - Message to send.
  • exclude_player number - Player index (1-16) to exclude.
-- Broadcast a message to every player except a specific one
-- Loops through player slots 1-16 (standard Halo CE server limit)
local function sendExclude(message, exclude_player)
    for i = 1, 16 do
        if player_present(i) and i ~= exclude_player then
            say(i, message)   -- 'say' sends message to individual player
        end
    end
end

-- Example: sendExclude("Server restart in 30 seconds!", 3) -- everyone except player 3

Determines if the player is currently inside a vehicle by reading the vehicle ID offset.

Parameters: dynamic_player number - Dynamic player memory address.

end

check_if_player_is_in_range

Determines if two 3D points are within a specified radius. Uses squared distance comparison for efficiency (avoids slow square root).

Parameters: x1,y1,z1, x2,y2,z2, radius number.

Returns: boolean - true if within radius.

-- Euclidean distance squared comparison
local function inRange(x1, y1, z1, x2, y2, z2, radius)
    local dx = x1 - x2
    local dy = y1 - y2
    local dz = z1 - z2
    return (dx * dx + dy * dy + dz * dz) <= (radius * radius)
end
-- Example: if inRange(px, py, pz, flagX, flagY, flagZ, 5.0) then ... end

check_if_vehicle_occupied

Checks whether a specific vehicle object is currently occupied by any player.

Parameters: vehicle_object number - Vehicle object memory address.

Returns: boolean - true if occupied.

-- Iterates all players and compares vehicle object pointer
local function isVehicleOccupied(vehicle_object)
    for i = 1, 16 do
        local dynamic_player = get_dynamic_player(i)
        if player_present(i) and player_alive(i) and dynamic_player ~= 0 then
            local vehicle_id = read_dword(dynamic_player + 0x11C)
            if vehicle_id == 0xFFFFFFFF then goto next_player end
            local vehicle_obj = get_object_memory(vehicle_id)
            if vehicle_obj ~= 0 and vehicle_obj == vehicle_object then
                return true
            end
        end
        ::next_player::
    end
    return false
end

check_player_invisibility_state

Determines if the specified player is currently invisible (active camouflage or admin invisibility).

Parameters: player_id number - Player index (1-16).

Returns: boolean - true if invisible.

-- Invisibility flag stored as float at offset 0x37C (1.0 = invisible)
local function isPlayerInvisible(player_id)
    local dynamic_player = get_dynamic_player(player_id)
    local invisible = read_float(dynamic_player + 0x37C)
    return dynamic_player ~= 0 and invisible == 1
end

custom_spawn_with_rotation

Writes position and rotation data to the dynamic object in memory, allowing precise teleportation with facing direction.

Parameters: dynamic_player, px, py, pz, pR, z_off (optional, default 0.3).

-- Rotates player using forward vector based on radians
local function spawnObject(dynamic_player, px, py, pz, pR, z_off)
    z_off = z_off or 0.3  -- vertical lift to avoid ground collision
    local x, y, z = px, py, pz + z_off
    write_vector3d(dynamic_player + 0x5C, x, y, z)                       -- position
    write_vector3d(dynamic_player + 0x74, math.cos(pR), math.sin(pR), 0) -- direction
end

deep_copy_table

Creates a deep copy of a table, including nested tables and metatables. Warning: may cause infinite recursion on self-referential tables.

Parameters: original_table any - The value or table to copy.

Returns: Fully independent copy.

local function deepCopy(original_table)
    if type(original_table) ~= "table" then return original_table end
    local copy = {}
    for key, value in pairs(original_table) do
        copy[deepCopy(key)] = deepCopy(value)   -- recursive copy for keys and values
    end
    return setmetatable(copy, deepCopy(getmetatable(original_table)))
end

find_tag_by_name_substring

Scans the global tag table for tags whose path/name contains a substring. Optional class filter ('weap','vehi','eqip').

local function findTagByNameSubstring(substring, class_filter)
    local base_tag_table = 0x40440000
    substring = substring:lower()
    local tag_array = read_dword(base_tag_table)
    local tag_count = read_dword(base_tag_table + 0xC)
    for i = 0, tag_count - 1 do
        local tag = tag_array + 0x20 * i
        local class = read_dword(tag)
        if class_filter == nil or class == read_dword(lookup_tag(class_filter, "")) then
            local name_ptr = read_dword(tag + 0x10)
            if name_ptr ~= 0 then
                local name = read_string(name_ptr)
                if name and name:lower():find(substring, 1, true) then
                    return read_dword(tag + 0xC) -- returns MetaIndex
                end
            end
        end
    end
    return nil
end

format_message_string Two approaches

Version 1: string.format based

local HELLO_MESSAGE = "Hello world!"
local PLAYER_JOINED = "Player %s has joined the game."

local function formatMessage(message, ...)
    if select('#', ...) > 0 then return message:format(...) end
    return message
end
-- Example: formatMessage(PLAYER_JOINED, "Chalwk")

Version 2: Placeholder‑based ($name style)

local SCORE_MESSAGE = "$name scored $points points."
local function formatMessage(message, vars)
    return (message:gsub("%$(%w+)", function(key)
        return vars[key] or "$" .. key
    end))
end
-- Example: formatMessage(SCORE_MESSAGE, {name = "Jay", points = 42})

get_flag_object_meta_and_tag_name

Finds the flag objective tag in the map and returns its meta ID and tag name.

local function getFlagData()
    local base_tag_table = 0x40440000
    local tag_array = read_dword(base_tag_table)
    local tag_count = read_dword(base_tag_table + 0xC)
    for i = 0, tag_count - 1 do
        local tag = tag_array + 0x20 * i
        local tag_class = read_dword(tag)
        if tag_class == 0x77656170 then -- "weap"
            local tag_data = read_dword(tag + 0x14)
            if read_bit(tag_data + 0x308, 3) == 1 then   -- objective weapon flag
                if read_byte(tag_data + 2) == 0 then     -- flag type = 0
                    return read_dword(tag + 0xC), read_string(read_dword(tag + 0x10))
                end
            end
        end
    end
    return nil, nil
end

get_objective_flag_or_oddball

Checks if the player's currently held weapon is an objective (oddball, flag, or any).

Parameters: dynamic_player, objective_type ("oddball","flag","any")

local function hasObjective(dynamic_player, objective_type)
    objective_type = objective_type or "any"
    local weapon_id = read_dword(dynamic_player + 0x118)
    local weapon_obj = get_object_memory(weapon_id)
    if weapon_obj == nil or weapon_obj == 0 then return false end
    local tag_address = read_word(weapon_obj)
    local tag_data_base = read_dword(0x40440000)
    local tag_data = read_dword(tag_data_base + tag_address * 0x20 + 0x14)
    if read_bit(tag_data + 0x308, 3) ~= 1 then return false end
    local obj_byte = read_byte(tag_data + 2)
    local is_oddball = (obj_byte == 4)
    local is_flag = (obj_byte == 0)
    return (objective_type == "oddball" and is_oddball) or
           (objective_type == "flag" and is_flag) or
           (objective_type == "any" and (is_oddball or is_flag))
end

get_player_inventory

Returns a table containing the player's weapons with ammo, clip, heat, and grenade counts.

local function getInventory(dynamic_player)
    local inventory = {}
    for i = 0, 3 do
        local weapon = read_dword(dynamic_player + 0x2F8 + i * 4)
        local object = get_object_memory(weapon)
        if object ~= 0 then
            inventory[i + 1] = {
                id = read_dword(object),                    -- weapon identifier
                ammo = read_word(object + 0x2B6),           -- primary ammo
                clip = read_word(object + 0x2B8),           -- bullets in current magazine
                ammo2 = read_word(object + 0x2C6),          -- secondary reserve (e.g., grenades for launcher)
                clip2 = read_word(object + 0x2C8),          -- secondary loaded ammo
                heat = read_float(object + 0x240),          -- overheating (0=cool, 1=overheated)
                frags = read_byte(dynamic_player + 0x31E),  -- frag grenade count
                plasmas = read_byte(dynamic_player + 0x31F) -- plasma grenade count
            }
        end
    end
    return inventory
end

get_player_world_coordinates

Retrieves player world position with crouch height adjustment and vehicle support.

local function getPlayerPosition(dynamic_player)
    local crouch = read_float(dynamic_player + 0x50C)
    local vehicle_id = read_dword(dynamic_player + 0x11C)
    local vehicle_obj = get_object_memory(vehicle_id)
    local x, y, z
    if vehicle_id == 0xFFFFFFFF then
        x, y, z = read_vector3d(dynamic_player + 0x5C)
    elseif vehicle_obj ~= 0 then
        x, y, z = read_vector3d(vehicle_obj + 0x5C)
    end
    local z_off = (crouch == 0) and 0.65 or 0.35 * crouch
    return x, y, z + z_off
end

map_object_scanner

Scans all weapon, vehicle, and equipment tags in the map and prints debug information.

local function getClassName(class)
    return class == 0x76656869 and "vehi"
        or class == 0x77656170 and "weap"
        or class == 1701931376 and "eqip"
end

local function scanMapObjects()
    local base_tag_table = 0x40440000
    local tag_array = read_dword(base_tag_table)
    local tag_count = read_dword(base_tag_table + 0xC)
    for i = 0, tag_count - 1 do
        local tag = tag_array + 0x20 * i
        local class = read_dword(tag)
        local class_name = getClassName(class)
        if class_name then
            local name_ptr = read_dword(tag + 0x10)
            local name = (name_ptr ~= 0) and read_string(name_ptr) or "no-name"
            local meta = read_dword(tag + 0xC)
            local tag_data = read_dword(tag + 0x14)
            if tag_data ~= 0 then
                cprint(class_name .. " meta=" .. meta .. " tag_data=0x" .. string.format("%X", tag_data) .. " name=" .. name, 12)
            end
        end
    end
end

parse_args

Splits a string into substrings based on a delimiter (space, comma, etc.). Ideal for command parsing.

local function parseArgs(input, delimiter)
    local result = {}
    for substring in input:gmatch("([^" .. delimiter .. "]+)") do
        result[#result + 1] = substring
    end
    return result
end
-- Example: parseArgs("kick 5 excessive spam", " ") -> {"kick","5","excessive","spam"}

set_respawn_time

Sets the respawn time for a player by writing directly to the player table memory (ticks = seconds * 33).

local function setRespawnTime(player_id, respawnTime)
    respawnTime = respawnTime or 3
    local static_player = get_player(player_id)
    if static_player then
        write_dword(static_player + 0x2C, respawnTime * 33)
    end
end

shuffle_table_fisher_yates

Randomizes the order of elements in an array-style table using the Fisher-Yates algorithm (in-place).

local function shuffleTable(tbl)
    -- Start from the end of the table and work backwards
    for i = #tbl, 2, -1 do
        -- Pick a random index from 1 to i
        local j = math.random(i)
        -- Swap the elements at positions i and j
        tbl[i], tbl[j] = tbl[j], tbl[i]
    end
end
-- Example: shuffleTable(players) -- random order of player names

vanish_player

Relocates a player off-map by modifying world coordinates in the static player table. Must be called every tick to maintain effect. Call in OnTick or a loop to keep player hidden.

local function vanish(player_id)
    -- Get the static player table address
    local static_player = get_player(playerId)
    if not static_player then return end

    -- Get the dynamic player object
    local dyn_player = get_dynamic_player(playerId)
    if dyn_player == 0 then return end

    -- Requires getPlayerPosition function from previous snippet
    local x, y, z = getPlayerPosition(dyn_player)
    if not x then return end

    -- Off-map offsets
    local x_off, y_off, z_off = -1000, -1000, -1000

    -- Relocate player off-map by writing new coordinates to player table
    write_float(static_player + 0xF8, x + x_off)
    write_float(static_player + 0xFC, y + y_off)
    write_float(static_player + 0x100, z + z_off)
end
